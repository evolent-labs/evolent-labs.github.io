        ```lua
        ---@class StepData
        ---@field currentStep number
        ---@field serverData table<string, any>           -- Arbitrary server-side state shared with the client
        ---@field missionId string
        ---@field classId number                          -- Contract class ID (e.g., D, C, B, etc.)
        ---@field contractId string
        ---@field leaderId number                         -- State ID of the contract leader
        ---@field plate string
        ---@field car CarData                             -- Vehicle model and label

        ---@class Step
        ---@field isComplete fun(data: StepData): boolean -- Required. Checks if the step is complete
        ---@field preparation? fun(data: StepData): void  -- Optional. Setup before step starts
        ---@field onComplete? fun(data: StepData): void   -- Optional. Cleanup when step completes

        ---@class ClientMission
        ---@field steps Step[]                            -- Sequence of mission steps
        ---@field canceled fun(contract: Contract): void  -- Called if mission is aborted
        ```
